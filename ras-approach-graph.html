function renderShip(options) { const { cx, stemY, loa, beam, standard, ras, label, rasSide, fill, stroke, dashed, opacity } = options; const halfBeam = beam / 2; const rasX = rasSide === "right" ? cx + halfBeam : cx - halfBeam; const rasY = stemY + ras; const standardY = stemY + standard; const rasLabelX = rasSide === "right" ? rasX + 5 : rasX - 5; const rasLabelAnchor = rasSide === "right" ? "start" : "end"; return ` ${makeLine( cx, stemY, cx, stemY + loa, stroke, 0.25, "2 2" )} ${makeLine( cx - halfBeam, standardY, cx + halfBeam, standardY, stroke, 0.25 )} ${makeLine( cx - halfBeam, rasY, cx + halfBeam, rasY, stroke, 0.25, "1.5 1.5" )} ${makeLine( cx - 1.6, rasY, cx + 1.6, rasY, stroke, 0.45 )} ${makeText( cx, stemY + loa + 7, label, 4.2, "middle", stroke )} ${makeText( cx, standardY - 2, "STANDARD", 3.1, "middle", stroke )} ${makeText( rasLabelX, rasY - 3, "RAS POINT", 3.1, rasLabelAnchor, stroke )} `; }
